    def self.define_meta_field(*fields)
      class_eval do
        fields.each do |field_name|
          define_method(field_name) do |*varg|
            if varg.size == 0
              instance_variable_get("@#{field_name}")
            else 
              instance_variable_set("@#{field_name}", *varg)
            end
          end
        end
      end
    end
