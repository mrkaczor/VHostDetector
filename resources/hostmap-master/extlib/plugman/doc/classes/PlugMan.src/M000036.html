    def PlugMan.start_all_plugins
      @logger.debug { "Starting all plugins" }
      @registered_plugins.keys.each do |name|
        begin
          start_plugin(name)
        rescue PluginError => err
          @logger.error { "Error starting plugin #{name.inspect}, reason #{err.error_type}" }
        end
      end
    end
